1.Builder Pattern

  • Pada program ini, class ConfigBuilder digunakan untuk menerapkan konsep builder pattern dalam pembuatan objek Config.
  • Method host() dan port() digunakan untuk mengatur nilai konfigurasi secara bertahap sebelum objek dibuat.
  • Setiap method pada class ConfigBuilder mengembalikan nilai this sehingga method dapat dipanggil secara berantai (method chaining).
  • Method build() digunakan untuk membuat dan mengembalikan objek Config setelah semua konfigurasi selesai diatur.
  • Pada fungsi main(), objek konfigurasi dibuat dengan cara memanggil method secara berurutan menggunakan konsep builder pattern.
  • Program kemudian menampilkan informasi host, port, dan pesan bahwa konfigurasi berhasil dibuat.

Kode Dart:


// ==========================================
// Nama  : Yoza Ulpia Rafila
// NIM   : 251410004
// Kelas : SI2A
// ==========================================

class Config {}

class ConfigBuilder {

  ConfigBuilder host(String h) {
    print("Host : $h");
    return this;
  }

  ConfigBuilder port(int p) {
    print("Port : $p");
    return this;
  }

  Config build() {
    print("Config berhasil dibuat");
    return Config();
  }
}

void main() {

  ConfigBuilder()
      .host('localhost')
      .port(8080)
      .build();
}

3.Solusi dengan Extension Methods

  • Pada program ini, extension methods digunakan untuk menambahkan method baru pada tipe data String tanpa mengubah class aslinya.
  • Method capitalize() digunakan untuk mengubah huruf pertama menjadi huruf kapital.
  • Getter isEmail digunakan untuk memeriksa apakah teks memiliki format email sederhana.
  • Method reverse() digunakan untuk membalik urutan karakter pada string.
  • Dengan menggunakan extension methods, pemanggilan method menjadi lebih singkat, rapi, dan mudah dibaca karena dapat langsung dipanggil dari objek string.
  • Program kemudian menampilkan hasil kapitalisasi, pengecekan email, dan pembalikan teks ke layar.

extension StringExtensions on String {

  String capitalize() {
    if (isEmpty) return this;
    return this[0].toUpperCase() + substring(1);
  }

  bool get isEmail => contains('@');

  String reverse() {
    return split('').reversed.join();
  }
}

void main() {

  var name = 'john';

  var capitalized = name.capitalize();
  var isEmail = 'test@mail.com'.isEmail;
  var reversed = 'hello'.reverse();

  print('Capitalized    : $capitalized');
  print('Is Email       : $isEmail');
  print('Reverse hello  : $reversed');
}

extension DateTimeExtensions on DateTime {

  String toFormattedString() {
    return '$day/$month/$year';
  }

  // Getter
  bool get isToday {
    var now = DateTime.now();

    return year == now.year &&
           month == now.month &&

extension NullableStringExtensions on String? {

  String orEmpty() => this ?? '';

  bool get isNullOrEmpty => this == null || this!.isEmpty;

  bool get isNotNullOrEmpty => !isNullOrEmpty;

  String withDefault(String defaultValue) {

extension NumberExtensions on num {

  // Format ke Rupiah
  String toRupiah() {
    return 'Rp${toStringAsFixed(0)}';
  }

  // Cek nilai dalam range
  bool isBetween(num min, num max) {
    return this >= min && this <= max;
  }

  // Pangkat
  num power(int exponent) {

    for (var i = 1; i < exponent; i++) {
      result *= this;
    }

    return result;
  }
}

8.Generic Extensions

  • Pada program ini, generic extension digunakan pada tipe data List<T> sehingga dapat digunakan untuk berbagai jenis data.
  • Method getOrNull() digunakan untuk mengambil data berdasarkan index secara aman tanpa menyebabkan error.
  • Method chunk() digunakan untuk membagi list menjadi beberapa bagian kecil sesuai ukuran yang ditentukan.
  • Method distinct() digunakan untuk menghapus data yang duplikat pada list.
  • Pada fungsi main(), program membagi list angka menjadi beberapa kelompok dan menghapus data yang sama.
  • Program kemudian menampilkan hasil pengolahan list ke layar.

Kode Dart:


// ==========================================
// Nama  : Yoza Ulpia Rafila
// NIM   : 251410004
// Kelas : SI2A
// ==========================================

extension ListExtensions on List {

  // Mengambil index dengan aman
  T? getOrNull(int index) {
    if (index < 0 || index >= length) {
      return null;
    }

    return this[index];
  }

  // Membagi list menjadi beberapa bagian
  List> chunk(int size) {

    var chunks = >[];

    for (var i = 0; i < length; i += size) {

      chunks.add(
        sublist(i, i + size > length ? length : i + size)
      );
    }

    return chunks;
  }

  // Menghapus data duplikat
  List distinct() {

    var result = [];

    for (var item in this) {

      if (!result.contains(item)) {
        result.add(item);
      }
    }

    return result;
  }
}

void main() {

  var numbers = [1, 2, 3, 4, 5, 6, 7];

  print('Data chunk : ${numbers.chunk(3)}');

  print(
    'Data distinct : ${[1, 2, 2, 3, 3].distinct()}'
  );
}

// ======================================
// Nama  : Yoza Ulpia Rafila
// NIM   : 251410004
// Kelas : SI.2A
// ======================================

class Product {

  final String name;
  final double price;

  Product(this.name, this.price);
}

// Extension format Rupiah
extension DoubleRupiahFormatter on double {

  String toRupiah() {

    final intValue = toInt();

    String numberString =
        intValue.toString();

    String formattedString = '';

    int counter = 0;

    for (int i =
            numberString.length - 1;
        i >= 0;
        i--) {

      formattedString =
          numberString[i] +
          formattedString;

      counter++;

      if (counter % 3 == 0 &&
          i != 0) {

        formattedString =
            '.$formattedString';
      }
    }

    return 'Rp$formattedString';
  }
}

// Extension Product
extension ProductExtensions on Product {

  // Method diskon
  Product withDiscount(
      double percentage) {

    var discountedPrice =
        price *
        (1 - percentage / 100);

    return Product(
      name,
      discountedPrice,
    );
  }

  // Getter informasi produk
  String get displayInfo =>
      '$name - ${price.toRupiah()}';

  // Getter cek harga mahal
  bool get isExpensive =>
      price > 100000;
}

void main() {

  var laptop =
      Product('Laptop', 15000000);

  var discounted =
      laptop.withDiscount(10);

  print("=== DATA LAPTOP ===");

  print(
    discounted.displayInfo
  );

  print(
    "Laptop mahal? "
    "${laptop.isExpensive}"
  );

  var keyboard =
      Product('Keyboard', 75000);

  print("\n=== DATA KEYBOARD ===");

  print(
    keyboard.displayInfo
  );

  print(
    "Keyboard mahal? "
    "${keyboard.isExpensive}"
  );
}

// NIM   : 251410004
// Kelas : SI.2A

// Extension validator String
extension StringValidators on String {

  // Validasi email
  bool get isEmail {

    return RegExp(
      r'^[\w\.-]+@([\w-]+\.)+[\w-]{2,4}$',
    ).hasMatch(this);
  }

  // Validasi nomor HP
  bool get isPhoneNumber {

    return RegExp(
      r'^[0-9]{10,13}$',
    ).hasMatch(this);
  }

  // Validasi password kuat
  bool get isStrongPassword {

    return length >= 8 &&
        contains(
          RegExp(r'[A-Z]')
        ) &&
        contains(
          RegExp(r'[a-z]')
        ) &&
        contains(
          RegExp(r'[0-9]')
        );
  }

  // Validasi URL
  bool get isUrl {

    try {

      Uri.parse(this);

      return true;

    } catch (_) {

      return false;
    }
  }
}

// Extension validator int
extension IntValidators on int {

  bool get isAdultAge =>
      this >= 18;

  bool get isPositive =>
      this > 0;

  bool get isEven =>
      this % 2 == 0;

  bool get isOdd =>
      this % 2 != 0;
}
